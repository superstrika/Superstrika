*** Installing Blinka ***
# 1. Update your system
'sudo apt-get update'
'sudo apt-get upgrade'

# 2. Install Python pip (if you don't have it)
'sudo apt-get install python3-pip'

# 3. Install Blinka (The hardware compatibility layer)
'pip3 install adafruit-blinka'

# 4. Install the specific VCNL4040 library
'pip3 install adafruit-circuitpython-vcnl4040'
